Table users {
  id uuid [primary key]
  email varchar [unique, not null]
  first_name varchar
  last_name varchar
  password varchar [not null]
  balance bigint [default: 0]
  profile_image varchar
  created_at timestamp
  updated_at timestamp
}

Table banner{
  id integer [primary key]
  banner_name varchar
  banner_image varchar
  description text
}

Table services{
  id integer [primary key]
  service_code varchar [unique, not null]
  service_name varchar
  service_icon varchar
  service_tariff bigint
}

table transactions{
  id uuid [primary key]
  user_id uuid 
  service_id integer
  invoice_number varchar [unique]
  total_amount bigint
  transaction_type varchar
  description text
  transaction_date timestamp
  createdOn timestamp
}

Ref transaction: transactions.user_id > users.id 
Ref transaction: transactions.service_id > services.id
